出題目的：

本題主要是想測試同學是否知道在 C/C++ 的字元處理方面要印出 \ 的符號需要使用 \\ 來跳脫。

解題方法：

把 Sample Output 直接複製到程式碼中，用搜尋取代方式把 \ 換成 \\ 即可通過。

解題所需要的關鍵：

跳脫字元。

解題程式碼：

#include <stdio.h>
int main(int argc, char *argv[])
{
    printf("                  ___           ___           ___           ___           ___           ___ \n");
    printf("      ___        /\\  \\         /\\  \\         /\\  \\         /\\  \\         /\\  \\         /\\__\\\n");
    printf("     /\\  \\      /::\\  \\       /::\\  \\       /::\\  \\        \\:\\  \\       /::\\  \\       /:/  /\n");
    printf("     \\:\\  \\    /:/\\ \\  \\     /:/\\:\\  \\     /:/\\:\\  \\        \\:\\  \\     /:/\\:\\  \\     /:/  / \n");
    printf("     /::\\__\\  _\\:\\~\\ \\  \\   /::\\~\\:\\  \\   /::\\~\\:\\  \\       /::\\  \\   /::\\~\\:\\  \\   /:/  /  \n");
    printf("  __/:/\\/__/ /\\ \\:\\ \\ \\__\\ /:/\\:\\ \\:\\__\\ /:/\\:\\ \\:\\__\\     /:/\\:\\__\\ /:/\\:\\ \\:\\__\\ /:/__/   \n");
    printf(" /\\/:/  /    \\:\\ \\:\\ \\/__/ \\:\\~\\:\\ \\/__/ \\/__\\:\\/:/  /    /:/  \\/__/ \\:\\~\\:\\ \\/__/ \\:\\  \\   \n");
    printf(" \\::/__/      \\:\\ \\:\\__\\    \\:\\ \\:\\__\\        \\::/  /    /:/  /       \\:\\ \\:\\__\\    \\:\\  \\  \n");
    printf("  \\:\\__\\       \\:\\/:/  /     \\:\\ \\/__/        /:/  /     \\/__/         \\:\\ \\/__/     \\:\\  \\ \n");
    printf("   \\/__/        \\::/  /       \\:\\__\\         /:/  /                     \\:\\__\\        \\:\\__\\\n");
    printf("                 \\/__/         \\/__/         \\/__/                       \\/__/         \\/__/\n");
    return 0;
}
